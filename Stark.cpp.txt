#include <iostream>
using namespace std;

int main()
{
	int a, b; // проміжок Х
	float h;
	cout << "Enter a = ";

	cout << "Enter b = ";
	cin >> b;
	cout << "Enter h = ";
	cin >> h;

    while (x <= b) {

        x += h;
	};
}
